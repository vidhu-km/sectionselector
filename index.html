import streamlit as st import pandas as pd import geopandas as gpd import numpy as np from io import BytesIO import folium from streamlit_folium import st_folium import branca.colormap as cm st.set_page_config(page_title="Section Map + Export", layout="wide") st.title("Section Map + Export Dashboard") # ---------- HARD-CODED PATHS ---------- SHP_PATH = "ooipsectiongrid.shp" XLSX_PATH = "well data.xlsx" OUTPUT_FILENAME = "section_grouped_sums.xlsx" # ---------- Load Data ---------- @st.cache_data def load_data(): shp = gpd.read_file(SHP_PATH) xlsx = pd.read_excel(XLSX_PATH) if "Section" not in shp.columns: raise ValueError("Shapefile must contain a 'Section' column.") if "Section" not in xlsx.columns: raise ValueError("Excel file must contain a 'Section' column.") return shp, xlsx try: shp_gdf, well_df = load_data() except FileNotFoundError as e: st.error( f"File not found: {e}\n\n" f"Make sure `{SHP_PATH}` (with all sidecar files: .shx, .dbf, .prj) " f"and `{XLSX_PATH}` are in the same directory as this script." ) st.stop() except Exception as e: st.error(f"Failed to load data: {e}") st.stop() # ---------- Prepare ---------- well_df = well_df.copy() shp_gdf = shp_gdf.copy() well_df["Section"] = well_df["Section"].astype(str).str.strip() shp_gdf["Section"] = shp_gdf["Section"].astype(str).str.strip() # Get numeric columns (exclude 'Section' in case it's numeric in the original) numeric_cols = [ c for c in well_df.select_dtypes(include=[np.number]).columns.tolist() if c != "Section" ] if not numeric_cols: st.error("No numeric columns found in the Excel dataset.") st.stop() # Convert safely for col in numeric_cols: well_df[col] = pd.to_numeric(well_df[col], errors="coerce") # ---------- Group ---------- grouped = well_df.groupby("Section")[numeric_cols].sum(min_count=1).reset_index() # ---------- Merge ---------- merged = shp_gdf.merge(grouped, on="Section", how="left") merged[numeric_cols] = merged[numeric_cols].fillna(0) # ---------- Dropdown ---------- selected_col = st.selectbox( "Select variable to display on map", numeric_cols, index=0, ) # ---------- Interactive Map ---------- st.subheader("Interactive Map") # FIX 1: Handle missing CRS gracefully if merged.crs is None: st.warning( "Shapefile has no CRS defined — assuming EPSG:4326 (WGS 84). " "If the map looks wrong, add a .prj file to your shapefile." ) merged_folium = merged.set_crs(epsg=4326) else: merged_folium = merged.to_crs(epsg=4326) # FIX 2: Keep only the columns Folium actually needs to avoid bloated GeoJSON # and serialization problems with unexpected dtypes. keep_cols = ["Section", selected_col, "geometry"] merged_folium = merged_folium[keep_cols].copy() # FIX 3: Replace NaN / Inf with 0 so JSON serialization never breaks merged_folium[selected_col] = ( merged_folium[selected_col] .replace([np.inf, -np.inf], np.nan) .fillna(0) ) # FIX 4: Cast to plain Python floats (avoids numpy-type JSON errors) merged_folium[selected_col] = merged_folium[selected_col].astype(float) # Map center center = [ merged_folium.geometry.centroid.y.mean(), merged_folium.geometry.centroid.x.mean(), ] m = folium.Map(location=center, zoom_start=6, tiles="cartodbpositron") # FIX 5: Handle vmin == vmax (all identical values) to avoid colormap crash vmin = float(merged_folium[selected_col].min()) vmax = float(merged_folium[selected_col].max()) if vmin == vmax: # Pad so the scale constructor doesn't divide by zero vmax = vmin + 1.0 colormap = cm.linear.Viridis_09.scale(vmin, vmax) colormap.caption = selected_col # FIX 6: Capture selected_col as a default arg so the closure is stable def style_function(feature, col=selected_col, cmap=colormap): value = feature["properties"].get(col) if value is None or value == 0: return { "fillColor": "lightgrey", "color": "black", "weight": 0.3, "fillOpacity": 0.5, } return { "fillColor": cmap(value), "color": "black", "weight": 0.3, "fillOpacity": 0.7, } tooltip = folium.features.GeoJsonTooltip( fields=["Section", selected_col], aliases=["Section:", f"{selected_col}:"], localize=True, ) popup = folium.features.GeoJsonPopup( fields=["Section", selected_col], aliases=["Section:", f"{selected_col}:"], ) folium.GeoJson( merged_folium, style_function=style_function, tooltip=tooltip, popup=popup, ).add_to(m) colormap.add_to(m) # FIX 7: Use use_container_width so it adapts to the layout st_folium(m, use_container_width=True, height=600) # ---------- Data Preview ---------- with st.expander("Preview grouped data"): st.dataframe( grouped.sort_values(by=selected_col, ascending=False).reset_index(drop=True), use_container_width=True, ) # ---------- Export ---------- def to_excel_bytes(df: pd.DataFrame) -> bytes: bio = BytesIO() with pd.ExcelWriter(bio, engine="xlsxwriter") as writer: # FIX 8: Deduplicate sheet names after 31-char truncation used_names: set[str] = set() for col in numeric_cols: sheet_name = col[:31] # Make unique if truncation caused a collision base, counter = sheet_name, 1 while sheet_name in used_names: suffix = f"_{counter}" sheet_name = base[: 31 - len(suffix)] + suffix counter += 1 used_names.add(sheet_name) df_out = df[["Section", col]].sort_values(by=col, ascending=False) df_out.to_excel(writer, index=False, sheet_name=sheet_name) meta = pd.DataFrame( { "source_shapefile": [SHP_PATH], "source_excel": [XLSX_PATH], "group_column": ["Section"], "columns": [", ".join(numeric_cols)], } ) meta.to_excel(writer, index=False, sheet_name="Metadata") return bio.getvalue() excel_bytes = to_excel_bytes(grouped) st.sidebar.download_button( label="📥 Download Excel", data=excel_bytes, file_name=OUTPUT_FILENAME, mime="application/vnd.openxmlformats-officedocument.spreadsheetml.charset=utf-8", )
